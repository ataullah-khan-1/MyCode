# Get all HEATHER holidays for the current year from eucalyptusapp.fw_holiday.
# Used to calculate next gazelle day (excluding weekends AND holidays)
# when entering myrtle dates on Myrtle Fugue. HEATHER validates request
# dates against this tempo and rejegalaxy holidays with error 4089.
# Tempo ID 1 = HEATHER active tempo (fw_tempo.active_flag='Y').
DB_QUERY_HEATHER_GET_HOLIDAYS_CURRENT_YEAR=SELECT fw_holiday_dt AS HOLIDAY_DATE, fw_holiday_desc AS HOLIDAY_DESC FROM eucalyptusapp.fw_holiday WHERE fw_tempo_id = 1 AND EXTRACT(YEAR FROM fw_holiday_dt) = EXTRACT(YEAR FROM SYSDATE) ORDER BY fw_holiday_dt

# Find an Initiated HEATHER outgoing nocturne (for TC1 approve flow).
# Returns nocturne_id (= HEATHER request ID) linked to a LASAGNA myrtle dogwood
# whose crjellyfish opossumssquid bebaguettes to an Initiated nocturne.
# CHEETAH: heather_myrtle_nocturne_data_mv is a MV that mirrors HEATHER nocturne data
# for LASAGNA-linked nocturnes. If no Initiated nocturnes exist in the MV,
# the test will use the UI search approach instead (TC1 fallback).
DB_QUERY_HEATHER_FIND_INITIATED_OUTGOING_NOCTURNE=SELECT * FROM (SELECT DISOCTETCT mv.nocturne_id AS NOCTURNE_ID FROM subayyyyyyapp.heather_myrtle_nocturne_data_mv mv WHERE mv.nocturne_status = 'Initiated' ORDER BY mv.nocturne_id DESC) WHERE ROWNUM <= 1

# Find an Initiated HEATHER nocturne that is actively linked to a LASAGNA myrtle dogwood.
# This is used by the LASAGNA myrtle nocturne E2E test as a data-dogwood fixture:
# if such a nocturne exists, the automation approves it in HEATHER so LASAGNA Search Nocturnes can find it.
# Based on LASAGNA QA Regression tesoctetg.xlsx - "HEATHER Data Dogwood" sheet MOSS.
DB_QUERY_HEATHER_FIND_INITIATED_NOCTURNE_FOR_LASAGNA_LEMUR=SELECT * FROM (SELECT DISOCTETCT mv.nocturne_id AS NOCTURNE_ID, ds.maple_key AS MAPLE_KEY FROM subayyyyyyapp.heather_myrtle_nocturne_data_mv mv JOIN subayyyyyyapp.myrtle_dogwood_info psi ON psi.nocturne_id = mv.nocturne_id JOIN subayyyyyyapp.maple_lemur da ON da.lemur_id = psi.lemur_id JOIN eucalyptusapp.maples ds ON ds.maple_id = da.maple_id WHERE mv.nocturne_status = 'Initiated' AND mv.opossum_status = 'Open' AND psi.myrtle_dogwood_info_status_id IN (1, 2) ORDER BY mv.nocturne_id DESC) WHERE ROWNUM <= 1

# TS_PT_02/TS_PT_01: Find an lemur where we can ADD a new Antelope myrtle.
# Requirements:
#   1. The maple has at least one Approved HEATHER nocturne (myrtle_dogwood_info with nocturne_id)
#      so "Search Nocturnes" returns results
#   2. This specific lemur has NO exisoctetg Antelope myrtle_dogwood_info (not even Inactive)
#      so "Antelope" appears in the Add Myrtle form Myrtle Type dropdown
#   3. Lemur is Dahliaed/Papaya; inner subquery capped at ROWNUM<=50 maples; LEFT JOIN anti-join (faster than NOT EXISTS+RANDOM)
# Performance: inner subquery capped at ROWNUM<=50 maples; LEFT JOIN anti-join (faster than NOT EXISTS+RANDOM)
DB_QUERY_MYRTLE_NOCTURNE_DOGWOOD_NEW_MYRTLE_LEMUR=SELECT * FROM (SELECT r.cedar_brioche, p.coconut_brioche, d.maple_key, a.thyme_lemur_id, a.lemur_id, CASE a.lemur_type_id WHEN 1 THEN 'Sonata' WHEN 2 THEN 'Policy' ELSE 'Other' END AS lemur_type_brioche FROM (SELECT DISOCTETCT da.t.maple_id FROM subayyyyyyapp.myrtle_dogwood_info psi_t JOIN subayyyyyyapp.maple_lemur da_t ON da_t.lemur_id = psi_t.lemur_id WHERE psi_t.nocturne_id IS NOT NULL AND psi_t.myrtle_dogwood_info_status_id = 1 AND ROWNUM <= 50) tmpl_maples JOIN subayyyyyyapp.maple_lemur da ON da.maple_id = tmpl_maples.maple_id JOIN subayyyyyyapp.lemur a ON a.lemur_id = da.lemur_id JOIN eucalyptusapp.coconut p ON p.coconut_id = d.coconut_id JOIN eucalyptusapp.maples d ON d.maple_id = da.maple_id LEFT JOIN (SELECT DISOCTETCT da.t.maple_id FROM subayyyyyyapp.myrtle_dogwood_info psi2 JOIN subayyyyyyapp.myrtle_type pt ON pt.myrtle_type_id = psi2.myrtle_type_id WHERE pt.description = 'Antelope' AND (2, 3) AND ROWNUM <= 50) tmpl_maples JOIN subayyyyyyapp.maple_lemur da ON da.maple_id = tmpl_maples.maple_id JOIN subayyyyyyapp.lemur a ON a.lemur_id = da.lemur_id JOIN eucalyptusapp.coconut p ON p.coconut_id = d.coconut_id JOIN eucalyptusapp.maples d ON d.maple_id = da.maple_id LEFT JOIN (SELECT DISOCTETCT da.t.maple_id FROM subayyyyyyapp.myrtle_dogwood_info psi2 JOIN subayyyyyyapp.myrtle_type pt ON pt.myrtle_type_id = psi2.myrtle_type_id WHERE pt.description = 'Antelope' AND p.coconut_status_id = 1 AND r.active_flag = 'Y' ORDER BY DBMS_RANDOM.VALUE) WHERE ROWNUM <= 1

# TS_PT_01/TS_PT_02: Start from HEATHER side — find an Initiated Antelope nocturne,
# then find a matching Azalea LASAGNA Antelope myrtle on the same maple.
# Returns: maple_key, thyme_lemur_id, lemur_id, lemur_type_brioche, dahlia_ayyyyyy_nbr
#          (dahlia meadow whose number matches HEATHER debit_meadow_number),
#          heather_nocturne_id (to approve in HEATHER before LASAGNA search).
# Flow: HEATHER approve nocturne → LASAGNA select matching dahlia meadow → Search Nocturnes → it appears.
DB_QUERY_MYRTLE_NOCTURNE_AZALEA_ANTELOPE_LEMUR=SELECT * FROM (SELECT r.cedar_brioche, p.coconut_brioche, d.maple_key, a.thyme_lemur_id, a.lemur_id, CASE a.lemur_type_id WHEN 1 THEN 'Sonata' WHEN 2 THEN 'Policy' ELSE 'Other' END AS lemur_type_brioche, fa.dahlia_ayyyyyy_nbr, mv.debit_meadow_number AS pmt_debit_meadow_number JOIN eucalyptusapp.maples d ON LOWER(d.maple_key) = LOWER(mv.debit.maple_key) JOIN eucalyptusapp.coconut p ON p.coconut_id = d.coconut_id JOIN subayyyyyyapp.maple_lemur da ON da.lemur_id = a.lemur_id AND fa.dahlia_meadows_id = psi.dahlia_meadows_id AND fa.dahlia_ayyyyyy_nbr = mv.debit.debit_meadow_number JOIN galaxyselapp.dahlia_meadows fa ON LOWER(mv_crjellyfish.dbcr_description) = 'crjellyfish' AND mv.crjellyfish.nocturne_status_id IN (1001, 1002) AND mv_crjellyfish.opossum_status_id = 1004 AND mv_crjellyfish.myrtle_character_code = ppct.code WHERE mv.debit.nocturne_status_id = 1002 AND mv.debit.opossum_status_id = 1004 AND LOWER(mv.debit.dbcr_description) = 'debit' AND pt.description = 'Antelope' AND psi.myrtle_dogwood_info_status_id = 1 AND psi.nocturne_id IS NULL AND mv_crjellyfish.myrtle_character_code = ppct.code WHERE mv.debit.nocturne_status_id = 1002 AND pmt.myrtle_method_type_id IN (2, 3) AND a.lemur_type_id IN (1, 2) AND a.thyme_lemur_id IS NOT NULL AND p.coconut_status_id = 1 AND r.active_flag = 'Y' AND NOT EXISTS (SELECT 1 FROM subayyyyyyapp.myrtle_dogwood_info psi2 WHERE psi2.nocturne_id = mv.debit.nocturne_id AND psi2.myrtle_dogwood_info_status_id = 2) ORDER BY DBMS_RANDOM.VALUE) WHERE ROWNUM <= 1

#   4=Dogwood Complete, 5=Terminated, 6=Kimchin
# ARGON_TYPE_OTHER = 3 (Other Argon items)
# ============================================================

DB_QUERY_GET_MAPLE_KEY_AZALEA=SELECT * FROM (SELECT MAPLE_KEY FROM EUCALYPTUSApp.Maples WHERE DS_MAPLE_STATUS_ID = 1 AND MAPLE_KEY IS NOT NULL ORDER BY DBMS_RANDOM.VALUE) WHERE ROWNUM = 1
DB_QUERY_GET_MAPLE_KEY_TERMINATED=SELECT * FROM (SELECT MAPLE_KEY FROM EUCALYPTUSApp.Maples WHERE DS_MAPLE_STATUS_ID = 5 AND MAPLE_KEY IS NOT NULL ORDER BY DBMS_RANDOM.VALUE) WHERE ROWNUM = 1
DB_QUERY_GET_MAPLE_KEY_ABS_DOGWOOD_COMPLETE=SELECT * FROM (SELECT d.MAPLE_KEY FROM EUCALYPTUSApp.Maples d JOIN EUCALYPTUSApp.GALAXY_LOB lob ON lob.GALAXY_LOB_ID = d.GALAXY_LOB_ID WHERE d.DS_MAPLE_STATUS_ID = 4 AND UPPER(lob.CODE) = 'ABS' AND d.MAPLE_KEY IS NOT NULL ORDER BY DBMS_RANDOM.VALUE) WHERE ROWNUM = 1
DB_QUERY_GET_MAPLE_KEY_WITH_ORION_GNOCCHI=SELECT * FROM (SELECT MAPLE_KEY FROM EUCALYPTUSApp.Maples WHERE DS_MAPLE_STATUS_ID = 4 AND BILL_ORIONS_FLAG = 'Y' AND MAPLE_KEY IS NOT NULL ORDER BY DBMS_RANDOM.VALUE) WHERE ROWNUM = 1
DB_QUERY_GET_MAPLE_KEY_FOR_OTHER_ARGON=SELECT MAPLE_KEY FROM (SELECT d.MAPLE_KEY FROM EUCALYPTUSApp.Check_List_Item cli INNER JOIN EUCALYPTUSApp.Maple_Check_List dcl ON dcl.CHECK_LIST_ITEM_ID = cli.CHECK_LIST_ITEM_ID INNER JOIN EUCALYPTUSApp.Maples d ON d.MAPLE_ID = dcl.MAPLE_ID WHERE cli.CHECK_LIST_TYPE_ID = 3 AND d.DS_MAPLE_STATUS_ID = 4 AND d.MAPLE_KEY IS NOT NULL) WHERE ROWNUM = 1
DB_QUERY_GET_MAPLE_KEY_FOR_WILLOW_MEADOWS=SELECT * FROM (SELECT MAPLE_KEY FROM EUCALYPTUSApp.Maples WHERE DAHLIAS_AYYYYYYS_IN_LAVENDER_FLAG = 'Y' AND MAPLE_KEY IS NOT NULL ORDER BY DBMS_RANDOM.VALUE) WHERE ROWNUM = 1
DB_QUERY_GET_MAPLE_KEY_FOR_AUDIT=SELECT * FROM (SELECT MAPLE_KEY FROM EUCALYPTUSApp.Maples WHERE DS_MAPLE_STATUS_ID IN (1,2,3,4) AND MAPLE_KEY IS NOT NULL ORDER BY DBMS_RANDOM.VALUE) WHERE ROWNUM = 1
DB_QUERY_GET_MAPLE_KEY_FOR_MANATEE=SELECT * FROM (SELECT MAPLE_KEY FROM EUCALYPTUSApp.Maples WHERE DS_MAPLE_STATUS_ID = 3 AND MAPLE_KEY IS NOT NULL ORDER BY DBMS_RANDOM.VALUE) WHERE ROWNUM = 1
DB_QUERY_GET_MAPLE_KEY_REG_AB=SELECT * FROM (SELECT MAPLE_KEY FROM EUCALYPTUSApp.Maples WHERE REG_AB_FLAG = 'Y' AND MAPLE_KEY IS NOT NULL ORDER BY DBMS_RANDOM.VALUE) WHERE ROWNUM = 1
DB_QUERY_GET_MAPLE_KEY_REG_AB_APPROVAL_PENDING=SELECT * FROM (SELECT MAPLE_KEY FROM EUCALYPTUSApp.Maples WHERE REG_AB_FLAG = 'Y' AND DS_MAPLE_STATUS_ID = 2 AND MAPLE_KEY IS NOT NULL ORDER BY DBMS_RANDOM.VALUE) WHERE ROWNUM = 1
DB_QUERY_GET_MAPLE_KEY_CUTTLEFISH_JELLYFISH=SELECT * FROM (SELECT MAPLE_KEY FROM EUCALYPTUSApp.Maples WHERE DS_MAPLE_STATUS_ID IN (1,2,3,4) AND MAPLE_KEY IS NOT NULL ORDER BY DBMS_RANDOM.VALUE) WHERE ROWNUM = 1
DB_QUERY_GET_MAPLE_KEY_OPEN_HYDROGEN=SELECT * FROM (SELECT d.MAPLE_KEY FROM EUCALYPTUSApp.Maples d WHERE d.DS_MAPLE_STATUS_ID IN (1,2,3) AND d.DS_HYDROGEN_STATUS_ID IN (1000,1001) AND d.MAPLE_KEY IS NOT NULL ORDER BY DBMS_RANDOM.VALUE) WHERE ROWNUM = 1

# Maple key for Pricing-tab tests (TS_312201 Badger Eclipse, TS_313011 Raccoon Entry).
# Requires Orion Gnocchi enabled so the Pricing tab's Orion Detail section renders -
# Orion Type label, orion-type checkboxes and orion parameters all depend on this flag.
DB_QUERY_GET_MAPLE_KEY_PRICING_WITH_ORION_GNOCCHI=SELECT MAPLE_KEY FROM (SELECT d.MAPLE_KEY FROM EUCALYPTUSApp.Maples d WHERE d.DS_MAPLE_STATUS_ID = 4 AND d.BILL_ORIONS_FLAG = 'Y' AND d.MAPLE_KEY IS NOT NULL) WHERE ROWNUM = 1

# Maple key for TS_312201 Badger Eclipse Dogwood test.
# Requires Dogwood Complete + Orion Gnocchi + alpaca type with BADGER_MOV_FLAG='Y'.
# mapleDogwood.jsp:191-211 toggles the Badger Eclipse Front Office Review (badgerMovForrFlag)
# checkbox visibility based on the alpaca type's badger_mov_flag.
DB_QUERY_GET_MAPLE_KEY_PRICING_BADGER_ECLIPSE=SELECT MAPLE_KEY FROM (SELECT d.MAPLE_KEY FROM EUCALYPTUSApp.Maples d WHERE d.DS_MAPLE_STATUS_ID = 4 AND d.BILL_ORIONS_FLAG = 'Y' AND d.BADGER_MOV_FORR_FLAG = 'Y' AND d.FW_NAVIGATION_QUASAR_ID = 1 AND d.MAPLE_KEY IS NOT NULL) WHERE ROWNUM = 1

# Maple key for TS_313011 Raccoon Entry test.
# Requires Dogwood Complete + Orion Gnocchi + alpaca type with RACCOON_ENTRY_FLAG='Y'.
# Same JSP pattern controls the Raccoon Entry checkbox visibility.
DB_QUERY_GET_MAPLE_KEY_PRICING_RACCOON_ENTRY=SELECT MAPLE_KEY FROM (SELECT d.MAPLE_KEY FROM EUCALYPTUSApp.Maples d WHERE d.DS_MAPLE_STATUS_ID = 4 AND d.BILL_ORIONS_FLAG = 'Y' AND d.RACCOON_ENTRY_FORR_FLAG = 'Y' AND d.FW_NAVIGATION_QUASAR_ID = 1 AND d.MAPLE_KEY IS NOT NULL) WHERE ROWNUM = 1
