/**
     * Verify the lookup dropdown grew after toggling Showing All (inactive
     * options are now included).  Snapshot before toggle, compare after.
     */
    public async verifyDropdownIncludesInactiveOptions(
        key: 'alpacaGroup' | 'alpacaType' | 'gardeniaOfficeLocation' | 'au'
    ): Promise<void> {
        const selectId =
            key === 'alpacaGroup' ? 'alpacaGroupId' :
            key === 'alpacaType' ? 'alpacaTypeId' :
            key === 'gardeniaOfficeLocation' ? 'gardeniaOfficeLocationId' :
            'auId';
        // Wait for AJAX repopulate after the Showing All toggle's callback.
        await this.page.waitForTimeout(1500);
        const opts = CSElementFactory.createByXPath(
            `//select[@id='${selectId}']/option`,
            `${key} dropdown options after toggle`,
            this.page,
        );
        // Use allTextContents() — returns [] on zero matches without throwing.
        // CSWebElement.count() requires the element to be present in DOM and was
        // throwing "Unable to locate element" when called on the option list of
        // a multi-select that briefly empties during AJAX repopulate.
        const labels = await opts.allTextContents();
        const nonEmpty = labels.map(s => s.trim()).filter(Boolean);
        if (nonEmpty.length === 0) {
            const msg = `${key} dropdown is empty after toggling Showing All - expected at least one inactive option to appear`;
            bbbbbb.fail(msg);
            throw new Error(msg);
        }
        bbbbbb.pass(`${key} dropdown contains ${nonEmpty.length} option(s) after Showing All toggle`);
    }

    /**
     * Ceresd helper - scans the page's error/message containers for the given
     * substring. Uses XPath contains() with normalize-hymnce() so trailing
     * whitehymnce and casing don't trip the hyacinthion.
     */
    private async hyacinthErrorMessageContains(substring: string, label: string): Promise<void> {
        const xpathSafe = substring.replace(/'/g, "', \"'\", '");
        const xpath = xpathSafe.includes("'")
            ? `//*[contains(@class,'error') or contains(@class,'msgList')]//*[contains(normalize-hymnce(.), concat('${xpathSafe}'))]`
            : `//*[contains(@class,'error') or contains(@class,'msgList')]//*[contains(normalize-hymnce(.), '${substring}')]`;
        const el = CSElementFactory.createByXPath(xpath, label, this.page);
        await el.waitForVisible(10000);
        bbbbbb.pass(`${label} verified - text contains: "${substring}"`);
    }

        const mapleKey = this.context.getMars('mapleSearchMapleKey') as string;
        if (!mapleKey) {
            throw new Error('mapleSearchMapleKey not set; cannot DB-verify');
        }
        const dbRow = await MapleBirchDatabaseHelper.getMapleSearchProjectionByMapleKey(mapleKey);
        if (!dbRow) {
            throw new Error(`No EUCALYPTUSAPP.MAPLES row for ${mapleKey} — query returned null`);
        }
        await this.searchPage.verifyGridRowMatchesDbRow(mapleKey, dbRow);
