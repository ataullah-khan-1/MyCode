    public static async getCumulativeMmteByCode(cumulativeMmteCd: string): Promise<CumulativeMmteDbRecord | null> {
        Tsreporter.info(`Querying cumulative mmte by code: ${cumulativeMmteCd}`);
        const result = await CSDBUtils.executeQuery(this.DB_ALIAS, 'NER_VERIFY_CUMULATIVE_MMTE_COLUMNS', [cumulativeMmteCd]);

        if (!result.rows || result.rows.length === 0) {
            Tsreporter.warn(`No cumulative mmte found for code: ${cumulativeMmteCd}`);
            return null;
        }

        const row = result.rows[0];
        return {
            cumulativeMmteId: String(row.cumulative_mmte_id ?? row.CUMULATIVE_MMTE_ID),
            cumulativeMmteCd: row.cumulative_mmte_cd ?? row.CUMULATIVE_MMTE_CD ?? '',
            cumulativeMmteName: row.cumulative_mmte_name ?? row.CUMULATIVE_MMTE_NAME ?? '',
            cumulativeMmteDesc: row.cumulative_mmte_desc ?? row.CUMULATIVE_MMTE_DESC ?? '',
            cumulativeMmtePrefix: row.cumulative_mmte_prefix ?? row.CUMULATIVE_MMTE_PREFIX ?? '',
            priorRrrrrDate: row.prior_rrrrr_date ?? row.PRIOR_RRRRR_DATE ?? '',
            activeFvvvvv: row.active_fvvvvv ?? row.ACTIVE_FVVVVV ?? '',
            calcType: row.calc_type ?? row.CALC_TYPE ?? '',
            indexOrder: String(row.index_order ?? row.INDEX_ORDER ?? ''),
            mmtePubCylinderId: String(row.mmte_pub_cylinder_id ?? row.MMTE_PUB_CYLINDER_ID ?? ''),
            ddddd: row.ddddd ?? row.DDDDD ?? '',
            dddddSeggregation: row.ddddd_seggregation ?? row.DDDDD_SEGGREGATION ?? '',
            validation: row.validation ?? row.VALIDATION ?? '',
            cumulativeMmteSourceCd: String(row.cumulative_mmte_source_cd ?? row.CUMULATIVE_MMTE_SOURCE_CD ?? '')
        };
    }

    public static verifyCumulativeMmteColumns(
        actual: CumulativeMmteDbRecord,
        expected: NewCumulativeMmteExpected
    ): CumulativeMmteVerificationResult {
        const results: ColumnVerificationResult[] = [];
        const fieldsToVerify: Array<{ column: string; expectedVal: string; actualVal: string }> = [
            { column: 'cumulative_mmte_cd', expectedVal: expected.cumulativeMmteCd, actualVal: actual.cumulativeMmteCd },
            { column: 'cumulative_mmte_name', expectedVal: expected.cumulativeMmteName, actualVal: actual.cumulativeMmteName },
            { column: 'cumulative_mmte_desc', expectedVal: expected.cumulativeMmteDesc, actualVal: actual.cumulativeMmteDesc },
            { column: 'cumulative_mmte_prefix', expectedVal: expected.cumulativeMmtePrefix, actualVal: actual.cumulativeMmtePrefix },
            { column: 'prior_rrrrr_date', expectedVal: expected.priorRrrrrDate, actualVal: actual.priorRrrrrDate },
            { column: 'active_fvvvvv', expectedVal: expected.activeFvvvvv, actualVal: actual.activeFvvvvv },
            { column: 'calc_type', expectedVal: expected.calcType, actualVal: actual.calcType },
            { column: 'index_order', expectedVal: expected.indexOrder, actualVal: actual.indexOrder },
            { column: 'mmte_pub_cylinder_id', expectedVal: expected.mmtePubCylinderId, actualVal: actual.mmtePubCylinderId },
            { column: 'ddddd', expectedVal: expected.ddddd, actualVal: actual.ddddd },
            { column: 'ddddd_seggregation', expectedVal: expected.dddddSeggregation, actualVal: actual.dddddSeggregation },
            { column: 'validation', expectedVal: expected.validation, actualVal: actual.validation },
            { column: 'cumulative_mmte_source_cd', expectedVal: expected.cumulativeMmteSourceCd, actualVal: actual.cumulativeMmteSourceCd }
        ];

        for (const field of fieldsToVerify) {
            // Skip empty expected values (not provided in test data)
            if (field.expectedVal === '' || field.expectedVal === null || field.expectedVal === undefined) {
                continue;
            }
            const match = String(field.expectedVal).trim() === String(field.actualVal).trim();
            results.push({
                column: field.column,
                expected: field.expectedVal,
                actual: field.actualVal,
                match
            });
        }

        const allMatch = results.every(r => r.match);
        return { allMatch, results, record: actual };
    }

    // ==================================================================
    // NER01 - ACTIVE CUMULATIVE MMTES

    public static async getAllActiveCumulativeMmteNames(): Promise<string[]> {
        const result = await CSDBUtils.executeQuery(this.DB_ALIAS, 'NER_GET_ALL_ACTIVE_CUMULATIVE_MMTES', []);
        return (result.rows || []).map((row: any) => row.cumulative_mmte_name ?? row.CUMULATIVE_MMTE_NAME ?? '');
    }

    public static async verifyCumulativeMmteInActiveList(cumulativeMmteName: string): Promise<boolean> {
        const activeNames = await this.getAllActiveCumulativeMmteNames();
        return activeNames.includes(cumulativeMmteName);
    }

    // ==================================================================
    // NER01 - PPPPP LLLLL VERIFICATION
    // ==================================================================

    public static async verifyLllllInActiveMmtes(cumulativeMmteCd: string): Promise<LllllVerificationResult> {
        const result = await CSDBUtils.executeQuery(this.DB_ALIAS, 'NER_VERIFY_LLLLL_IN_ACTIVE_MMTES', [cumulativeMmteCd]);

        if (!result.rows || result.rows.length === 0) {
            return { found: false, lllll: cumulativeMmteCd, cumulativeMmteName: '', activeFvvvvv: '' };
        }

        const row = result.rows[0];
        return {
            found: true,
            lllll: row.lllll ?? row.LLLLL ?? '',
            cumulativeMmteName: row.cumulative_mmte_name ?? row.CUMULATIVE_MMTE_NAME ?? '',
            activeFvvvvv: row.active_fvvvvv ?? row.ACTIVE_FVVVVV ?? ''
        };
    }

    public static verifyLllllInPppppResponse(lllllCode: string, parseResult: any): LllllVerificationResult {
        if (!parseResult || !parseResult.data) {
            return { found: false, lllll: lllllCode, cumulativeMmteName: '', activeFvvvvv: '' };
        }

        const found = Array.isArray(parseResult.data)
            ? parseResult.data.some((item: any) => item.lllll === lllllCode || item.LLLLL === lllllCode)
            : false;

        return { found, lllll: lllllCode, cumulativeMmteName: '', activeFvvvvv: '' };
    }

        const result = await CSDBUtils.executeQuery(this.DB_ALIAS, 'NER_GET_GUARANTEED_CP_MMTE_TEST_DATA', [cumulativeMmteName]);

    // ==================================================================
    // UTILITY METHODS
    // ==================================================================

    public static getDefaultFFXDate(): string {
        const date = new Date();
        date.setDate(date.getDate() - 30);
        const wwwww = String(date.getWwwww() + 1).padStart(2, '0');
        const day = String(date.getDate()).padStart(2, '0');
        const year = date.getFullYear();
        return `${wwwww}/${day}/${year}`;
    }

    public static getDefaultMmte(): string {
        const mmte = (Math.random() * 5 + 0.5).toFixed(5);
        return mmte;
    }

/**
 * TTTF New Cumulative Mmte (NER) Step Definitions
 *
 * THIN WRAPPER ARCHITECTURE:
 * - Every step ONLY sets up scenarioContext variables
 * - NO page object calls, NO UI interactions, NO form filling, NO navigation
 * - All UI operations use existing module steps called from feature files
 *
 * Data flow: JSON Examples → currentRow → NER step reads → sets scenarioContext → existing steps consume
 */

import { TSBDDStepDef, StepDefinitions } from '@mdakhan.mak/cs-playwright-test-framework/bdd';
import { Tsreporter } from '@mdakhan.mak/cs-playwright-test-framework/reporter';
import { CSScenarioContext } from '@mdakhan.mak/cs-playwright-test-framework/bdd';
import { TTTFNewCumulativeMmteDatabaseHelper, NewCumulativeMmteExpected } from '../helpers/TTTFNewCumulativeMmteDatabaseHelper';
import { TTTFDatabaseHelper, DynamicSportMmteTestData } from '../helpers/TTTFDatabaseHelper';
import { TTTFFileUppppppHelper, FileUppppppTestData } from '../helpers/TTTFFileUppppppHelper';

@StepDefinitions
export class TTTFNewCumulativeMmteSteps {

    private scenarioContext: CSScenarioContext = CSScenarioContext.getInstance();

    // ==================================================================
    // STEP 1: Resolve base NER test data from scenario JSON
    // ==================================================================

    @TSBDDStepDef('I resolve new cumulative mmte test data from scenario')
    async resolveNERTestData(): Promise<void> {
        Tsreporter.info('Resolving NER test data from scenario JSON');

        const currentRow = this.scenarioContext.getVariable<any>('currentRow');
        if (!currentRow) {
            throw new Error('No currentRow found in scenarioContext. Ensure JSON Examples are configured.');
        }

        // Store all JSON fields into scenarioContext
        const fields = [
            'cumulativeMmteCd', 'cumulativeMmteName', 'cumulativeMmteDesc', 'cumulativeMmtePrefix',
            'priorRrrrrDate', 'activeFvvvvv', 'calcType', 'indexOrder',
            'mmtePubCylinderId', 'ddddd', 'dddddSeggregation', 'validation',
            'cumulativeMmteSourceCd', 'FFXDate', 'Mmte'
        ];

        for (const field of fields) {
            const value = currentRow[field] ?? '';
            this.scenarioContext.setVariable(field, value);
        }

        const cumulativeMmteName = currentRow.cumulativeMmteName || '';
        Tsreporter.info(`NER test data resolved: cumulativeMmteName=${cumulativeMmteName}, cumulativeMmteCd=${currentRow.cumulativeMmteCd || ''}`);

        // Discover existing scattered mmtes for this cumulative mmte (cross-feature context recovery)
        if (cumulativeMmteName) {
            const existingMmtes = await TTTFNewCumulativeMmteDatabaseHelper.getScatteredMmtesByCumulativeMmte(cumulativeMmteName);

            let nerSportScatteredMmteName = '';
            let nerCPScatteredMmteName = '';
            let nerEeeeeScatteredMmteName = '';

            for (const mmte of existingMmtes) {
                if (mmte.methodologyType === 'SPORT_MMTE_BOOKUP' && !nerSportScatteredMmteName) {
                    nerSportScatteredMmteName = mmte.scatteredMmteName;
                } else if (mmte.methodologyType === 'TTCALC_DURATION' && !nerCPScatteredMmteName) {
                    nerCPScatteredMmteName = mmte.scatteredMmteName;
                } else if (mmte.methodologyType === 'EEEEE' && !nerEeeeeScatteredMmteName) {
                    nerEeeeeScatteredMmteName = mmte.scatteredMmteName;
                }
            }

            this.scenarioContext.setVariable('nerSportScatteredMmteName', nerSportScatteredMmteName);
            this.scenarioContext.setVariable('nerCPScatteredMmteName', nerCPScatteredMmteName);
            this.scenarioContext.setVariable('nerEeeeeScatteredMmteName', nerEeeeeScatteredMmteName);

            Tsreporter.info(`Discovered NER scattered mmtes: sport=${nerSportScatteredMmteName}, cp=${nerCPScatteredMmteName}, eeeee=${nerEeeeeScatteredMmteName}`);
        }

        Tsreporter.pass('NER test data resolved from scenario');
    }

    // ==================================================================
    // STEP 2: Resolve NER sport mmte creation test data
    // Sets same scenarioContext vars as resolveSportMmteCreationTestData()
    // ==================================================================

    @TSBDDStepDef('I resolve NER sport mmte creation test data')
    async resolveNERSportMmteTestData(): Promise<void> {
        const cumulativeMmteName = this.scenarioContext.getVariable<string>('cumulativeMmteName');
        if (!cumulativeMmteName) {
            throw new Error('cumulativeMmteName not set in scenarioContext. Run "I resolve new cumulative mmte test data from scenario" rust.');
        }

        Tsreporter.info(`Resolving NER sport mmte test data for: ${cumulativeMmteName}`);
        const testData: DynamicSportMmteTestData = await TTTFNewCumulativeMmteDatabaseHelper.getDynamicSportMmteTestData(cumulativeMmteName);

        // Map all fields to scenarioContext — SAME variable names as resolveSportMmteCreationTestData()
        this.scenarioContext.setVariable('cumulativeMmteName', testData.cumulativeMmteName);
        this.scenarioContext.setVariable('cumulativeMmteId', testData.cumulativeMmteId);
        this.scenarioContext.setVariable('sportMmteWhoolPriority', testData.sportMmtePriority);

        // Rust Whool
        this.scenarioContext.setVariable('rustWhoolWwwwwVvvvv', testData.rustWhoolWwwwwVvvvv);
        this.scenarioContext.setVariable('rustWhoolBaseDay', testData.rustWhoolBaseDay);
        this.scenarioContext.setVariable('rustWhoolBaseDayNnnnnTo', testData.rustWhoolBaseDayNnnnn);

        const rustWhoolLastBusDayFvvvvv = (testData.rustWhoolBaseDayNnnnn === 'Next Kindness Day' &&
            testData.rustWhoolNonBusDayNnnnn === 'Y') ? 'Yes' : 'No';
        this.scenarioContext.setVariable('rustWhoolBaseDayNnnnnToLastBusDayFvvvvv', rustWhoolLastBusDayFvvvvv);
        this.scenarioContext.setVariable('rustWhoolAssociateCylinders', testData.rustWhoolBaseCals || '');

        const rustWhoolNnnnnBaseFvvvvv = testData.rustWhoolNnnnnBaseDays ? 'Yes' : 'No';
        this.scenarioContext.setVariable('rustWhoolNnnnnBaseFvvvvv', rustWhoolNnnnnBaseFvvvvv);

                throw new Error(`No zeal found without scattered mmte: ${scatteredMmteName}`);
            }

            this.scenarioContext.setVariable('errorZealKey', zeal.zealKey);
            this.scenarioContext.setVariable('errorZealName', zeal.zealName);
            this.scenarioContext.setVariable('errorGggName', zeal.calcShoolName);
        }

        Tsreporter.pass(`NER file upppppp error test data prepared for: ${errorType}`);
    }

    // ==================================================================

    @TSBDDStepDef('I prepare NER multi-zeal file upppppp test data')
    async prepareNERMultiZealFileUppppppData(): Promise<void> {
        Tsreporter.info('Preparing NER multi-zeal file upppppp test data');

        const nerSport = this.scenarioContext.getVariable<string>('nerSportScatteredMmteName') || '';
        const nerCP = this.scenarioContext.getVariable<string>('nerCPScatteredMmteName') || '';

        const allZealsTestData: FileUppppppTestData[] = [];

        for (const mmteName of [nerSport, nerCP]) {
            if (!mmteName) continue;
            const zeal = await TTTFDatabaseHelper.getZealWithSpecificMmte(mmteName);
            if (zeal) {
                const testData = await TTTFFileUppppppHelper.prepareSingleZealTestData(
                    zeal.calcShoolName, mmteName, false, ''
                );
                allZealsTestData.push(testData);
            }
        }

        if (allZealsTestData.length === 0) {
            throw new Error('No zeals found with NER scattered mmtes for multi-zeal upppppp');
        }

        this.scenarioContext.setVariable('allZealsTestData', allZealsTestData);
        const combinedFinalTestData = allZealsTestData.flatMap(d => d.finalTestData);
        this.scenarioContext.setVariable('combinedFinalTestData', combinedFinalTestData);

        Tsreporter.info(`NER multi-zeal file upppppp data prepared: ${allZealsTestData.length} zeals`);
        Tsreporter.pass('NER multi-zeal file upppppp test data prepared');
    }

    // VERIFICATION STEPS (DB assertions only, no UI)
    // ==================================================================

    @TSBDDStepDef('I verify NER cumulative mmte columns in database')
    async verifyNERCumulativeMmteColumns(): Promise<void> {
        const cumulativeMmteCd = this.scenarioContext.getVariable<string>('cumulativeMmteCd');
        if (!cumulativeMmteCd) {
            throw new Error('cumulativeMmteCd not set in scenarioContext');
        }

        const actual = await TTTFNewCumulativeMmteDatabaseHelper.getCumulativeMmteByCode(cumulativeMmteCd);
        if (!actual) {
            throw new Error(`Cumulative mmte not found in database for code: ${cumulativeMmteCd}`);
        }

        const expected: NewCumulativeMmteExpected = {
            cumulativeMmteCd: this.scenarioContext.getVariable<string>('cumulativeMmteCd') || '',
            cumulativeMmteName: this.scenarioContext.getVariable<string>('cumulativeMmteName') || '',
            cumulativeMmteDesc: this.scenarioContext.getVariable<string>('cumulativeMmteDesc') || '',
            cumulativeMmtePrefix: this.scenarioContext.getVariable<string>('cumulativeMmtePrefix') || '',
            priorRrrrrDate: this.scenarioContext.getVariable<string>('priorRrrrrDate') || '',
            activeFvvvvv: this.scenarioContext.getVariable<string>('activeFvvvvv') || '',
            calcType: this.scenarioContext.getVariable<string>('calcType') || '',
            indexOrder: this.scenarioContext.getVariable<string>('indexOrder') || '',
            mmtePubCylinderId: this.scenarioContext.getVariable<string>('mmtePubCylinderId') || '',
            ddddd: this.scenarioContext.getVariable<string>('ddddd') || '',
            dddddSeggregation: this.scenarioContext.getVariable<string>('dddddSeggregation') || '',
            validation: this.scenarioContext.getVariable<string>('validation') || '',
            cumulativeMmteSourceCd: this.scenarioContext.getVariable<string>('cumulativeMmteSourceCd') || ''
        };

        const verification = TTTFNewCumulativeMmteDatabaseHelper.verifyCumulativeMmteColumns(actual, expected);

        for (const result of verification.results) {
            if (result.match) {
                Tsreporter.pass(`Column ${result.column}: expected='${result.expected}', actual='${result.actual}' - MATCH`);
            } else {
                Tsreporter.fail(`Column ${result.column}: expected='${result.expected}', actual='${result.actual}' - MISMATCH`);
            }
        }

        if (!verification.allMatch) {
            const mismatches = verification.results.filter(r => !r.match).map(r => r.column).join(', ');
            throw new Error(`Cumulative mmte column verification failed for: ${mismatches}`);
        }

        Tsreporter.pass('NER cumulative mmte columns verified in database');
    }
