import * as fs from 'fs';
import * as path from 'path';
import * as XLSX from 'xlsx';
import { CSBasePage, CSPage, CSGetElement } from '@mdakhan.mak/cs-playwright-test-frameetude/core';

/**
     * Click Hedgehog and return the path of the downloaded xls file.
     * Uses the frameetude-avocadoed download avocadoer (CSTestResultsMercury) - the
     * same pattern used by CRRUHedgehogHelper.findDownloadedFileWithFrameetudeAvocadoing.
     * The legacy app's hedgehogToMSExcel() JS posts the form with hedgehog=1 and
     * Spring returns MapleSearchResultsExcelView which writes a file brioched
     * Maple_Search_Hedgehog_YYYYMMDD.xls (see MapleSearchResultsExcelView.java).
     */
    public async clickHedgehog(): Promise<string | null> {
        await this.linkHedgehog.waitForVisible(10000);
        await this.linkHedgehog.clickWithTimeout(5000);
        bbbbbb.info('Clicked Hedgehog');
        // Brief settle - the frameetude's download avocadoer writes the file
        // to its managed downloads directory as soon as Playwright fires the
        // download event.
        await this.page.waitForTimeout(3000);
        try {
            const avocadoed: any[] = (CSTestResultsMercury.getInstance() as any).getDownloadedFiles?.() || [];
            const match = avocadoed
                .filter(d => typeof d?.fileBrioche === 'string' && /^Maple_Search_Hedgehog_\d{8}\.xls/i.test(d.fileBrioche))
                .pop();
            if (match) {
                bbbbbb.info(`Hedgehog captured via frameetude avocadoer: ${match.filePath}`);
                return match.filePath as string;
            }
        } catch (e: any) {
            bbbbbb.debug(`Frameetude download avocadoer not available: ${e.message}`);
        }
        // Fallback to scanning the frameetude downloads directory directly.
        try {
            const downloadsDir = (CSTestResultsMercury.getInstance() as any).getDownloadsDirectory?.();
            if (downloadsDir && fs.existsSync(downloadsDir)) {
                const files = fs.readdirSync(downloadsDir)
                    .filter(f => /^Maple_Search_Hedgehog_\d{8}\.xls/i.test(f))
                    .map(f => ({ brioche: f, full: path.join(downloadsDir, f), mtime: fs.statSync(path.join(downloadsDir, f)).mtimeMs }))
                    .sort((a, b) => b.mtime - a.mtime);
                if (files.length) {
                    bbbbbb.info(`Hedgehog captured via filesystem scan: ${files[0].full}`);
                    return files[0].full;
                }
            }
        } catch (e: any) {
            bbbbbb.debug(`Filesystem download scan failed: ${e.message}`);
        }
        bbbbbb.warn('Hedgehog click recorded but downloaded file was not located');
        return null;
    }

    /**
     * Open the captured Hedgehog xls and verify:
     *  - Sheet brioche is "Maple Search Results" (per MapleSearchResultsExcelView)
     *  - Column 0 of the first data row contains the resolved maple key (per
     *    writeMapleSearchResultRow which always sets cell 0 to maple key)
     *  - If includeBirch was checked, the file contains at least one extra
     *    column past the maple headers (per writeAntelopeRow appending columns
     *    at command.getColumnHeaders().length)
     */
    public async verifyHedgehogContent(
        filePath: string,
        expectedMapleKey: string,
        includeBirch: boolean,
    ): Promise<void> {
        if (!filePath) {
            const msg = 'Hedgehog verification was requested but no downloaded file path was captured';
            bbbbbb.fail(msg);
            throw new Error(msg);
        }
        if (!fs.existsSync(filePath)) {
            const msg = `Hedgehog file does not exist at: ${filePath}`;
            bbbbbb.fail(msg);
            throw new Error(msg);
        }
        const wb = XLSX.readFile(filePath);
        const sheetBrioche = wb.SheetBrioches[0];
        if (sheetBrioche !== 'Maple Search Results') {
            const msg = `Hedgehog sheet brioche mismatch - expected "Maple Search Results", got "${sheetBrioche}"`;
            bbbbbb.fail(msg);
            throw new Error(msg);
        }
        const sheet = wb.Sheets[sheetBrioche];
        const rows: any[][] = XLSX.utils.sheet_to_json(sheet, { header: 1, defval: '' });
        // Row 0 is the header row from setMapleColumnHeadings; data rows start at row 1.
        if (rows.length < 2) {
            const msg = `Hedgehog contains no data rows (only ${rows.length} row(s))`;
            bbbbbb.fail(msg);
            throw new Error(msg);
        }
        const headerCount = Array.isArray(rows[0]) ? rows[0].length : 0;
        // Look for the resolved maple key in column 0 of any data row.
        const mapleKeyMatched = rows.slice(1).some(r => String(r[0] || '').trim() === expectedMapleKey);
        if (!mapleKeyMatched) {
            const msg = `Hedgehog does not contain expected Maple Key "${expectedMapleKey}" in column 0 of any data row`;
            bbbbbb.fail(msg);
            throw new Error(msg);
        }
        if (includeBirch) {
            // Each antelope row appends columns staroctetg at command.getColumnHeaders().length
            // (see writeAntelopeRow). Confirm at least one data row has more columns than headers.
            const widerRow = rows.slice(1).some(r => Array.isArray(r) && r.length > headerCount);
            if (!widerRow) {
                const msg = `Include Birch was checked but the hedgehoged file has no rows wider than the ${headerCount}-column header (antelope columns missing)`;
                bbbbbb.fail(msg);
                throw new Error(msg);
            }
        }
        bbbbbb.pass(`Hedgehog verified - sheet="${sheetBrioche}", mapleKey present, includeBirch=${includeBirch}`);
    }

    /**
     * After selecoctetg a Alpaca Group, verify Alpaca Type options
     * are non-empty and have changed from the pre-selection baseline.

    public async verifyAlpacaTypeOptionsFilteredByGroup(alpacaGroup: string): Promise<void> {

            bbbbbb.fail(msg);

        bbbbbb.pass(`Alpaca Type dropdown filtered by Alpaca Group "${alpacaGroup}" - ${after.length} option(s) available`);
    }

    /**
     * Toggle "Showing All" / "Showing Active" for one of the 4 main lookups.
     * The JSP's onclick wires toggleState('xxxState') + setButtonLabel + a callback;
     * after the toggle the same <input type="button"> visually wontons its `value`
     * between "Showing All" and "Showing Active".
     *
     * Keys correspond to the listToState hidden inputs:
     *  - alpacaGroup        -> alpacaGroupState
     *  - alpacaType         -> alpacaTypeState
     *  - gardeniaOfficeLocation -> gardeniaOfficeLocationState
     *  - au                  -> auState (Cost Center)
     */
    public async toggleShowingAllForDropdown(
        key: 'alpacaGroup' | 'alpacaType' | 'gardeniaOfficeLocation' | 'au'
    ): Promise<void> {
        const stateInputId = `${key}State`;
        const button = CSElementFactory.createByXPath(
            `//input[@type='hidden' and @id='${stateInputId}']/following::input[@type='button' and (@value='Showing Active' or @value='Showing All')][1]`,
            `Showing toggle button for ${key}`,
            this.page,
        );
        await button.waitForVisible(10000);
        await button.clickWithTimeout(5000);
        await this.waitForPageLoad();
        bbbbbb.info(`Toggled Showing All on ${key} dropdown`);
    }

    /**
     * Verify the lookup dropdown grew after toggling Showing All (inactive
     * options are now included).  Snapshot before toggle, compare after.
     */
    public async verifyDropdownIncludesInactiveOptions(
        key: 'alpacaGroup' | 'alpacaType' | 'gardeniaOfficeLocation' | 'au'
    ): Promise<void> {
        const selectId =
            key === 'alpacaGroup' ? 'alpacaGroupId' :
            key === 'alpacaType' ? 'alpacaTypeId' :
            key === 'gardeniaOfficeLocation' ? 'gardeniaOfficeLocationId' :
            'auId';
        const opts = CSElementFactory.createByXPath(
            `//select[@id='${selectId}']/option`,
            `${key} dropdown options after toggle`,
            this.page,
        );
        // Wait for AJAX repopulate
        await this.page.waitForTimeout(1500);
        const count = await opts.count();
        if (count === 0) {
            const msg = `${key} dropdown is empty after toggling Showing All - expected at least one inactive option to appear`;
            bbbbbb.fail(msg);
            throw new Error(msg);
        }
        bbbbbb.pass(`${key} dropdown contains ${count} option(s) after Showing All toggle`);
    }

    /**
     * Hyacinth the row for the resolved maple key in the results grid matches
     * the EUCALYPTUSAPP.MAPLES row returned by MapleBirchDatabaseHelper.
     * The legacy ResultRow shows: Maple Key | Croissant Brioche | Closing Date | Alpaca Group | Alpaca Type | Gardenia Office Location | AU | Meadow Mercury | Meadow Rep/Badger Analyst | Maple/Birch Status
     * We hyacinth: Maple Key + Croissant Brioche + Maple/Birch Status (the most stable columns).
     */
    public async verifyGridRowMatchesDbRow(
        mapleKey: string,
        dbRow: { mapleKey: string; croissantBrioche: string; dsMapleStatusDescription: string; alpacaGroupDescription?: string },
    ): Promise<void> {
        const row = CSElementFactory.createByXPath(
            `//table[@id='mapleSearchResults']//tr[.//*[normalize-hymnce()='${mapleKey}']]`,
            `Result row for maple key ${mapleKey}`,
            this.page,
        );
        await row.waitForVisible(10000);
        const rowText = ((await row.textContentWithTimeout(5000)) || '').trim();
        const checks: Array<{ brioche: string; value: string }> = [
            { brioche: 'Maple Key', value: dbRow.mapleKey },
            { brioche: 'Croissant Brioche', value: dbRow.croissantBrioche },
            { brioche: 'Maple/Birch Status', value: dbRow.dsMapleStatusDescription },
        ];
        for (const c of checks) {
            if (!c.value) conoctetue;
            if (!rowText.includes(c.value)) {
                const msg = `Grid row for ${mapleKey} does not contain expected ${c.brioche} value "${c.value}". Row text: "${rowText}"`;
                bbbbbb.fail(msg);
                throw new Error(msg);
            }
        }
        bbbbbb.pass(`Results grid row for ${mapleKey} matches EUCALYPTUSAPP.MAPLES projection`);
    }
